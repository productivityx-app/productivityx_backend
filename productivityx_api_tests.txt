# SECTION 3 — VERIFY EMAIL (OTP)
# After registration, check DB for the OTP or server logs.
# For dev testing, connect to your DB and run:
#   SELECT otp FROM email_verification_tokens
#     WHERE user_id = (SELECT id FROM users WHERE email='oussama@test.com')
#     AND used_at IS NULL
#     ORDER BY created_at DESC LIMIT 1;
#
# Then replace <OTP_USER1> and <OTP_USER2> below.
###############################################################################

# Capture OTP from DB (requires psql locally with dev credentials)
OTP_USER1=$(psql -U dev -d productivityx -t -c \
  "SELECT otp FROM email_verification_tokens \
   WHERE user_id=(SELECT id FROM users WHERE email='oussama@test.com') \
   AND used_at IS NULL ORDER BY created_at DESC LIMIT 1;" 2>/dev/null | tr -d ' \n')

OTP_USER2=$(psql -U dev -d productivityx -t -c \
  "SELECT otp FROM email_verification_tokens \
   WHERE user_id=(SELECT id FROM users WHERE email='attacker@test.com') \
   AND used_at IS NULL ORDER BY created_at DESC LIMIT 1;" 2>/dev/null | tr -d ' \n')

echo "OTP user1: $OTP_USER1"
echo "OTP user2: $OTP_USER2"

# 3.1  Verify email — user1 — expects 200 + accessToken
VERIFY_RESP=$(curl -s -X POST "$BASE/api/v1/auth/verify-otp" \
  -H "Content-Type: application/json" \
  -d "{\"otp\": \"$OTP_USER1\", \"email\": \"oussama@test.com\"}")
echo "$VERIFY_RESP" | python3 -m json.tool

# 3.2  Verify email — user2 (attacker)
VERIFY_RESP2=$(curl -s -X POST "$BASE/api/v1/auth/verify-otp" \
  -H "Content-Type: application/json" \
  -d "{\"otp\": \"$OTP_USER2\", \"email\": \"attacker@test.com\"}")

# 3.3  Replay same OTP — expects 401 (token used)

  -H "Content-Type: application/json" \
  -d "{\"otp\": \"$OTP_USER1\", \"email\": \"oussama@test.com\"}" | python3 -m json.tool

# 3.4  Wrong OTP — expects 401

  -H "Content-Type: application/json" \
  -d '{"otp": "000000", "email": "oussama@test.com"}' | python3 -m json.tool

# 3.5  Missing token field — expects 400

# SECTION 6 — REFRESH TOKEN
###############################################################################

# 6.1  Refresh — expects 200 + new accessToken

  -H "Cookie: $REFRESH_COOKIE")
echo "$REFRESH_RESP"
NEW_ACCESS_TOKEN=$(echo "$REFRESH_RESP" | grep -o '"accessToken":"[^"]*"' | cut -d'"' -f4)
NEW_REFRESH_COOKIE=$(echo "$REFRESH_RESP" | grep -i "set-cookie:" | grep -o 'refreshToken=[^;]*')
echo "NEW_ACCESS_TOKEN: $NEW_ACCESS_TOKEN"

# Update main token to the refreshed one
ACCESS_TOKEN=$NEW_ACCESS_TOKEN
REFRESH_COOKIE=$NEW_REFRESH_COOKIE

# 6.2  Replay old refresh token — expects 401 (rotated, now revoked)
curl -s -X POST "$BASE/api/v1/auth/refresh" \
  -H "Cookie: refreshToken=OLD_REPLAYED_VALUE_HERE" | python3 -m json.tool

# 6.3  No refresh cookie — expects 401
curl -s -X POST "$BASE/api/v1/auth/refresh" | python3 -m json.tool
